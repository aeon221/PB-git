#Convert A Province
obj_convert_a_province = {
	type = character
	
	creation_effect = {
		change_stewardship = -2
	}
	allow = {
		any_realm_province = {
			NOT = { religion = ROOT }
		}
		is_adult = yes
		is_ruler = yes
		NOT = { trait = incapable }
		NOT = { trait = cynical }
		piety = 0
		OR = {
			ai = no
			any_demesne_province = {
				NOT = { religion = ROOT }
			}
		}
	}
	chance = {
		factor = 100
		
		modifier = {
			factor = 1.25
			trait = patient
		}
		modifier = {
			factor = 1.25
			trait = diligent
		}
		modifier = {
			factor = 0.01
			trait = content
		}
		modifier = {
			factor = 10
			trait = zealous
		}
		modifier = {
			factor = 2.0
			trait = crusader
		}
		modifier = {
			factor = 0.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 1.5
			learning = 10
		}
	}
	success = {
		has_character_flag = converted_a_province
	}
	abort = {
		NOT = {
			any_realm_province = {
				NOT = { religion = ROOT }
			}
		}
	}
	abort_effect = {
		change_stewardship = 2
	}
	effect = {
		clr_character_flag = converted_a_province
		piety = 25
		change_stewardship = 2
		if = {
			limit = { NOT = { has_character_flag = converted_province_ambition_done } }
			change_learning = 1
		}
		set_character_flag = converted_province_ambition_done
		letter_event = { id = meneth.206 }
	}
}
